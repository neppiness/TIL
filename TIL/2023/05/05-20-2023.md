```java
    for (int i = 2; i <= 3; i++) {
        for (int j = i; j <= 10_000; j++) {
            for (int k = 1; k <= i; k++) {
                if (j - k < 0) continue;
                calc[i][j] += calc[k][j - i];
            }
        }
    }
```
